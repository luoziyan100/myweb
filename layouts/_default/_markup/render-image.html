{{/* Render hook for Markdown images - Normalizes local paths to respect baseURL with subdir (e.g., /myweb) - Leaves fully qualified http(s) URLs untouched */}} {{- $src := .Destination | safeURL -}} {{- if or (strings.HasPrefix $src "http://") (strings.HasPrefix $src "https://") -}} {{- else -}} {{- $norm := strings.TrimPrefix "/" $src -}} {{- end -}}
